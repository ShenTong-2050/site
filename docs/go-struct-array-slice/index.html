package main

import "fmt"

type Member struct {
    Id      int
    Name    string
    Age     int
}

func main() {

    // 使用方式
    var m1 = Member{Id:1,Name: "张三",Age: 18}	// 使用方式二

    var m2 Member
    m2.Id = 2
    m2.Name = "李四"
    m2.Age = 18

    // 使用方式三
    m3 := Member{Id: 3,Name: "王五",Age: 18}

    // 访问字段
    fmt.Println(m1.Id,m1.Name ,m1 .Age,m2 ,m3)

}

package main

import (
    "encoding/json"
    "fmt"
)

type Member struct {
    Id   int    `json:"json_id,omitempty"`
    Name string `json:"json_name,omitempty"`
    Age  int    `json:"json_age,omitempty"`
}

func construct(id,age int, name string) *Member {
    return &Member{
        Id:   id,
        Name: name,
        Age:  age,
    }
}

func main() {
    b,_ := json.Marshal(construct(4,53,"李四"))
    fmt.Println(string(b))
}

}

package slice

import (
    "fmt"
)

func MainSlice() {
    var (
        a []int					// nil 切片 和 nil 相等，一般用来表示一个不存在的切片
        b = []int{}				// 空切片 和 nil 不相等， 一般用来表示一个空的集合
        c = []int{1,2,3}		// 有 3 个元素的切片，len和cap都为3
        d = c[:2]				// 有 2 个元素的切片，len为2，cap为3
        e = c[0:2:cap(c)]		// 有 2 个元素的切片，len为2，cap为3
        f = c[:0]				// 有 0 个元素的切片，len为0，cap为3
        g = make([]int,3)		// 有 3 个元素的切片，len和cap都为3
        h = make([]int,2,3)		// 有 2 个元素的切片，len为2，cap为3
        i = make([]int,0,3)		// 有 0 个元素的切片，len为0，cap为3
    )
    fmt.Println(a,b,c,d,e,f,g,h,i)
}

小结：在底层数组指针、长度、容量没有发生变化的情况下，对切片的遍历、元素的读取、修改都和数组一样，在对切片赋值和传递时与数组类似，只是复制切片头信息，并不会复制底层的数据

与数组区别：切片的类型和长度无关，只要是相同类型元素构成的切片均对应相同的切片类型；

package main

import "fmt"

func main() {

   //var a = make([]int,100)
    var a []int
    
    // 尾部追加
    a = append(a,1)          // 追加 1 个元素
    a = append(a,2,3,4)       // 追加 3 个元素
    a = append(a,[]int{5,6,7}...)  // 追加 1 个切片
    
    fmt.Println(a)
    fmt.Println("开区间:",a[:2])
    fmt.Println("闭区间:",a[2:])
    
    // 头部追加
    a = append([]int{0},a...)                    // 在开头添加 1 个元素
    fmt.Println(a)
    a = append([]int{-3,-2,-1},a...)               // 在开头添加 1 个切片
    fmt.Println(a)
    
    // 在索引为 4 的位置插入 8
    // 第二个 append操作会构建一个临时切片,然后追加到 a[:4] 其实它的本质是 "截取"
    a = append(a[:4],append([]int{8},a[4:]...)...)    // 在索引为 4 的位置插入 8 (本质是截取之后在构建)
    fmt.Println(a)
    a = append(a[:4],append([]int{9,10},a[4:]...)...)  // 在索引为 4 的位置插入切片
    fmt.Println(a)
    
    // 可以通过 copy 与 append 组合来避免临时切片的创建,同样是插入操作
    a = append(a,0)       // 给原始切片插入一个空元素(截取的规则是右闭左开)
    copy(a[4+1:],a[4:])          // 将 a[4:] 向后移动 1 个位置
    a[4] = 11              // 修改切片元素的值
    fmt.Println(a)
    
    // 也可以在指定位置插入多个元素
    var tmpSlice = []int{12,13}
    a = append(a,tmpSlice...)        // 将 tmpSlice 所有元素追加至 a 切片后方
    copy(a[4+len(tmpSlice):],a[4:])       // 将 a[4:] 后边所有元素向后移动 len(tmpSlice) 个位置
    copy(a[4:],tmpSlice)            // 将 tmpSlice 拷贝到指定位置
    fmt.Println(a)
}

func delSlice() {

   var a = []int{1,2,3,4}

   // 第一种:移动指针的方式

   // 删除最后 1 个元素
   fmt.Println(a[:len(a)-1])           // [1 2 3]
   // 删除最后 2/N 个元素
   fmt.Println(a[:len(a)-2])           // [1 2]
   // 删除开头 1 个元素
   fmt.Println(a[1:])                // [2 3 4]
   // 删除开头 2/N 个元素
   fmt.Println(a[2:])                // [3 4]

   // 第二种:通过append原地完成,不会导致内存空间结构发生变化

   b := []int{5,6,7,8,9}

   // 删除开头 1 个元素
   fmt.Println(append(b[:0],b[1:]...))       // [6 7 8 9]
   // 删除开头 2/N 个元素
   fmt.Println(append(b[:0],b[2:]...))       // [7 8 9]

   // 删除切片中间的元素
   c := []int{1,2,3,4,5,6}
   c = append(c[:3],c[3+1:]...)         // 删除切片中间 1 个索引为 3 的元素 [1 2 3 5 6]
   //fmt.Println(c)
   c = append(c[:3],c[3+2:]...)         // 删除切片中间 2 个元素,索引分别为 3、4 [1 2 3 6]
   fmt.Println(c)
}

总结：切片高效操作的要点是降低内存分配的次数，要尽量保证 “append” 操作不会超出 “cap” 的容量，降低触发内存分配的次数和每次分配内存的大小
