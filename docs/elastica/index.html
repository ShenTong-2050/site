function __autoload_elastica ($class) {

创建/删除 索引

class Es {
    protected Client $elastica_client;
    protected string $index_name = 'test_index_name';
    public function __construct() {
        $this->elastica_client = new Client(array(
            'host' => '127.0.0.1',
            'port' => '9200',
        ));
    }
    
    /**
     * 创建索引
     */
    public function createIndex() {
        $elasticaIndex = $this->elastica_client->getIndex($this->index_name);
        // 索引 设置
        $indexParam['settings'] = array(
            'number_of_shards' => 3,
            'number_of_replicas' => 2,
            'analysis' => array(
                'analyzer' => array(
                    'ik_analyzer' => array(
                        'type' => 'ik_max_word',
                        'tokenizer' => 'ik_smart'
                    ),
                )
            )
        );
        $elasticaIndex->create($indexParam);
        echo 'success';
    }
    
    /**
     * 删除索引
     */
    public function deleteIndex() {
        $this->elastica_client->getIndex($this->index_name)->delete();
        echo 'success';
    }
}

type 和 tokenizer 在 ik_analyzer 配置中的含义【低版本中可用】：

type: 指定整体分词策略

ik_analyzer 支持多种类型:

  • ik_max_word:最大化切分,生成关键词。
  • ik_smart:最少切分,结合语义判定。
  • ik_lucene:不进行智能分析,直接使用 Lucene Standard 分词器。

这三种类型决定分词的细粒度和效果。

tokenizer：指定使用那种分词器，主要有：

  • ik_smart：使用 IKAnalyzer 的智能分词实现。
  • lucene：使用 Lucene 的 StandardTokenizer 分词器。

type 和 tokenizer 这两个参数在最新版本的Elasticsearch中不再被支持，分词器(analyzer)的配置方式在Elasticsearch7.x及更高版本中有较大改变:

  • 不再区分type和tokenizer这两个参数
  • 直接使用tokenizer参数指定分词器的实现
array(
    'analyzer' => array(
        'ik_analyzer' => array(
            'tokenizer' => 'ik_max_word',
        ),
    )
);

创建 Mapping

/**
 * 创建 Mapping
 */
public function createMapping() {
    $properties = array(
        'uuid'      => array('type' => 'keyword'),
        'name'      => array('type' => 'keyword'),
        'title'     => array(
            'type'  => 'keyword',
            'analyzer' => 'ik_analyzer',     // 指定分词器
            'store'        => true,         // 保留重复 keyword,默认删除重复项
            'ignore_above' => 3,            // keyword 长度大于 3 的字符会被保留，但位置会移动到末尾
        ),
        'age'       => array('type' => 'integer'),
        'phone'     => array('type' => 'keyword'),
        'color'     => array('type' => 'keyword'),
        'email'     => array('type' => 'keyword'),
        'company'   => array(
            'type' => 'text',
        ),
        'bank_card'     => array('type' => 'keyword'),
        'bank_name'     => array('type' => 'keyword'),
        'add_time'      => array('type' => 'date','format' => 'yyyy-MM-dd HH:mm:ss'),
    );
    $map = new Mapping();
    $map->setSource(['enabled' => true]);
    $map->setProperties($properties);
    $map->send($this->elastica_client->getIndex($this->index_name));
    echo "success";
}

/**
 * 添加Mapping
 */
public function addField() {
    $this->index_name = 'test_index';
    $index = $this->elastica_client->getIndex($this->index_name);
    $mapping = new Mapping([
        'tag'=>['type'=>'keyword']
    ]);
    $index->setMapping($mapping);
    echo 'success';
}

refresh操作：刷新指定索引,让最近写入该索引的文档更新立即搜索到。

由于 Elasticsearch 是基于 Lucene 的,默认情况下它不会实时将新添加或修改的文档更新到搜索索引中。它是采用背压策略,等待一定数量的更新后,再执行优化并刷新至索引中。这可以提高整体性能。所以,如果你刚向索引添加或修改了文档,然后立即搜索,这些更新的文档是搜索不到的。这时你就需要使用 refresh 操作,手动刷新整个索引,让这些最新更新被搜索到。

refresh 操作是非常耗费资源的,所以 Elasticsearch 并不鼓励经常使用。只有在刚写入更新,立即需要搜索结果时,才使用。在生产环境中,不要频繁调用 refresh，只有需要立即搜索最新结果时,才调用一次 refresh

搜索文档

  • Elastica\Search 可以用其中类的方法来设置查询参数、执行搜索查询、处理搜索结果、聚合数据等。

  • Elastica\Query 类 是 Elasticsearch 查询 API 的 PHP 封装。可以通过它构建各种类型的查询对象以便在 Elasticsearch 中执行搜索查询。

// 创建 Search 对象
$search = new Search($this->elastica_client);

$search -> addIndex($this->index_name);              // 指定查询的索引

// 构建查找条件
$query = new Query();
$query   ->setQuery($queryType)                     // 指定查找类型
         ->setFrom(0)                               // 设定offset
        ->setSize(10)                              // 记录数
        ->addSort(['id' => ['order' => 'desc']])   // 指定排序字段 与 排序方式
        ->setHighlight([
            'fields' => [
                'name' => new \stdClass()
            ],
            'pre_tags' => ['<em>'],
            'post_tags' => ['</em>'],              // 给 指定 查询结果的 字段 加上 em 标签
        ]);
        
$search -> search($query);                           // 执行查询
$search -> getResults();                             // 获取查找结果集对象
$search -> count();                                  // 获取查找结果的数量
$search -> getDocuments();                           // 获取文档集对象

\stdClass 是 PHP 中的一个内置类，它可以用来创建一个空对象。当您需要创建一个对象来存储一些属性和值，但不想使用自定义的类时，可以使用 \stdClass。

location 字段声明为 geo_point ，经纬度信息 可以是 字符串、数组 或 对象

数组格式：经度在前、纬度在后【lon、lat】
在 Elasticsearch 内部 不管是 字符串 还是 数组 都是 经度在前、纬度在后

multiMatch 多字段匹配

// multiMatch 多字段匹配文本
$queryType = new Query\MultiMatch();
$queryType->setQuery('忆戏');                                // 匹配的字符串
$queryType->setFields(['title','content']);                 // 匹配的字段
$queryType->setOperator(Query\MultiMatch::OPERATOR_AND);    // 匹配方式
$queryType->setMinimumShouldMatch(2);                       // 设置最小匹配数
// $queryType->setTieBreaker(0.3);                          // 调整评分权重
$queryType->setType(Query\MultiMatch::TYPE_PHRASE);         // 匹配类型,默认是 best_fields
// $queryType->setFuzziness(Query\MultiMatch::FUZZINESS_AUTO);    // 模糊匹配程度

匹配类型

  • best_fields：默认类型,匹配多个字段,并选择分数最高的那个字段作为结果。
  • most_fields：匹配多个字段,将多个字段的分数累加作为最终分数。
  • cross_fields：在多个字段间搜索,一个查询词可能出现在多个字段中。
  • phrase：短语查询,需要完全匹配查询短语。
  • phrase_prefix：短语前缀查询,需要匹配查询短语的前缀。

$queryType = new Query\MultiMatch();
$queryType->setQuery('回 忍');                                     // 匹配的字符串
$queryType->setFields(['content','title']);                       // 匹配的字段
$queryType->setOperator(Query\MultiMatch::OPERATOR_AND);          // 匹配方式
$queryType->setType(Query\MultiMatch::TYPE_BEST_FIELDS);          // 匹配类型,默认是 best_fields

问题

MultiMatch

text 类型用于全文搜索、keyword 类型适用于精确查询和过滤/聚合

分组查询

Elasticsearch 中的聚合查询是基于 Mapping（映射）实现的，因此对于不同类型的 Mapping 聚合也有一些限制：

  • 字段的数据类型必须是支持聚合操作的类型，例如 数字、日期、字符串 等。

  • 如果该字段的类型是字符串类型，那么该字段必须是未经分析的（not analyzed），即该字段不应该被分词器进行分词处理。如果该字段是经过分析的（analyzed），那么在进行分组操作时，Elasticsearch 会默认使用该字段的分词结果进行分组，这可能会导致聚合结果不准确。

  • 如果一个字段的数据类型是 嵌套类型（nested），那么就无法直接对该字段进行聚合操作需要使用嵌套聚合（Nested Aggregation => \Elastica\Aggregation\Nested）来实现。

分词器

从 Elasticsearch 8.0 开始已经删除了手动指定字段分词器的功能。

具体而言,Elasticsearch 8.x 引入了 Auto-mapping 特性,自动根据字段数据选择合适的分词器。 也就意味着,用户从 8.x 开始不再需要（也不允许）手动指定字段的分词器了【版本差异】
