// 注册自动加载机制
public static function register($autoload = '')
{
    // 注册系统自动加载 函数 autoload
    spl_autoload_register($autoload ?: 'think\\Loader::autoload', true, true);

    $rootPath = self::getRootPath();

    // 获取 composer 目录路径
    self::$composerPath = $rootPath . 'vendor' . DIRECTORY_SEPARATOR . 'composer' . DIRECTORY_SEPARATOR;

    // Composer自动加载支持
    if (is_dir(self::$composerPath)) {
        // 顶级命名空间初始化类，用于给核心类初始化顶级命名空间
        // ComposerStaticInite178cf12f908bc9f3e221ce38771c57e
        // 防止用户自定义类名跟这个类重复冲突了，所以在类名上加了一个 hash 值
        if (is_file(self::$composerPath . 'autoload_static.php')) {
            require self::$composerPath . 'autoload_static.php';

            // 返回 由 当前脚本中 已定义 类的名字 组成的数组
            $declaredClass = get_declared_classes();
            $composerClass = array_pop($declaredClass);
            foreach (['prefixLengthsPsr4', 'prefixDirsPsr4', 'fallbackDirsPsr4', 'prefixesPsr0', 'fallbackDirsPsr0', 'classMap', 'files'] as $attr) {
                if (property_exists($composerClass, $attr)) {
                    // 此处 是 将 composer 中 各映射规范文件 加载到本类 对应的规范的属性中
                    self::${$attr} = $composerClass::${$attr};
                }
            }
        } else {
            self::registerComposerLoader(self::$composerPath);
        }
    }


    // 注册命名空间
    self::addNamespace([
        'think'  => __DIR__,
        'traits' => dirname(__DIR__) . DIRECTORY_SEPARATOR . 'traits',
    ]);

    // 加载类库映射文件，提高系统自动加载的性能
    if (is_file($rootPath . 'runtime' . DIRECTORY_SEPARATOR . 'classmap.php')) {
        self::addClassMap(__include_file($rootPath . 'runtime' . DIRECTORY_SEPARATOR . 'classmap.php'));
    }

    // 自动加载extend目录
    self::addAutoLoadDir($rootPath . 'extend');
}

// 注册composer自动加载
public static function registerComposerLoader($composerPath)
{
    // 符合 PSR0 标准的自动加载文件，存放着顶级命名空间与文件的映射
    if (is_file($composerPath . 'autoload_namespaces.php')) {
        $map = require $composerPath . 'autoload_namespaces.php';
        foreach ($map as $namespace => $path) {
            self::addPsr0($namespace, $path);
        }
    }

    //符合 PSR4 标准的自动加载文件，存放着顶级命名空间与文件的映射
    if (is_file($composerPath . 'autoload_psr4.php')) {
        $map = require $composerPath . 'autoload_psr4.php';
        foreach ($map as $namespace => $path) {
            self::addPsr4($namespace, $path);
        }
    }

    // 自动加载的最简单形式，有完整的命名空间和文件目录的映射
    if (is_file($composerPath . 'autoload_classmap.php')) {
        $classMap = require $composerPath . 'autoload_classmap.php';
        if ($classMap) {
            self::addClassMap($classMap);
        }
    }

    //用于加载全局函数的文件，存放各个全局函数所在的文件路径名
    if (is_file($composerPath . 'autoload_files.php')) {
        self::$files = require $composerPath . 'autoload_files.php';
    }
}

// 自动加载
public static function autoload($class)
{
    // self::$classAlias 是一个类内使用的数组，此处是 判断当前类 的 别名是否存在
    if (isset(self::$classAlias[$class])) {
        // class_alias(原始类,类的别名)
        return class_alias(self::$classAlias[$class], $class);
    }

    // 通过类库文件、PSR-0 与 PSR-4 查找文件
    // http://static.kancloud.cn/xiak/php-node/191121
    // 此处 $file 为 false 或 文件地址
    if ($file = self::findFile($class)) {
        // Win环境严格区分大小写
        if (strpos(PHP_OS, 'WIN') !== false && pathinfo($file, PATHINFO_FILENAME) != pathinfo(realpath($file), PATHINFO_FILENAME)) {
            return false;
        }

        __include_file($file);
        return true;
    }
}

// 注册命名空间
    public static function addNamespace($namespace, $path = '')
    {
//        Array
//        (
//            [think] => /www/wwwroot/thinkphp5.1/thinkphp/library/think
//            [traits] => /www/wwwroot/thinkphp5.1/thinkphp/library/traits
//        )

        if (is_array($namespace)) {
            foreach ($namespace as $prefix => $paths) {
                self::addPsr4($prefix . '\\', rtrim($paths, DIRECTORY_SEPARATOR), true);
            }
        } else {
            self::addPsr4($namespace . '\\', rtrim($path, DIRECTORY_SEPARATOR), true);
        }
    }
    
    // 添加Psr4空间
    private static function addPsr4($prefix, $paths, $prepend = false)
    {
        if (!$prefix) {
            // Register directories for the root namespace.
            if ($prepend) {
                self::$fallbackDirsPsr4 = array_merge(
                    (array) $paths,
                    self::$fallbackDirsPsr4
                );
            } else {
                self::$fallbackDirsPsr4 = array_merge(
                    self::$fallbackDirsPsr4,
                    (array) $paths
                );
            }
        } elseif (!isset(self::$prefixDirsPsr4[$prefix])) {
            // Register directories for a new namespace.
            $length = strlen($prefix);
            if ('\\' !== $prefix[$length - 1]) {
                throw new \InvalidArgumentException("A non-empty PSR-4 prefix must end with a namespace separator.");
            }
    
            self::$prefixLengthsPsr4[$prefix[0]][$prefix] = $length;
            self::$prefixDirsPsr4[$prefix]                = (array) $paths;
        } elseif ($prepend) {
            // Prepend directories for an already registered namespace.
            self::$prefixDirsPsr4[$prefix] = array_merge(
                (array) $paths,
                self::$prefixDirsPsr4[$prefix]
            );
        } else {
            // Append directories for an already registered namespace.
            self::$prefixDirsPsr4[$prefix] = array_merge(
                self::$prefixDirsPsr4[$prefix],
                (array) $paths
            );
        }
    }
